631 /* Maximum size of the ASN.1 encoding of an INTEGER with the specified
632  * number of bits.
633  *
634  * This definition assumes that bits <= 2^19 - 9 so that the length field
635  * is at most 3 bytes. The length of the encoding is the length of the
636  * bit string padded to a whole number of bytes plus:
637  * - 1 type byte;
638  * - 1 to 3 length bytes;
639  * - 0 to 1 bytes of leading 0 due to the sign bit.
640  */
641 #define PSA_KEY_EXPORT_ASN1_INTEGER_MAX_SIZE(bits) ((bits) / 8 + 5)
642 
643 /* Maximum size of the export encoding of an RSA public key.
644  * Assumes that the public exponent is less than 2^32.
645  *
646  * RSAPublicKey ::= SEQUENCE {
647  * modulus INTEGER, -- n
648  * publicExponent INTEGER } -- e
649  *
650  * - 4 bytes of SEQUENCE overhead;
651  * - n : INTEGER;
652  * - 7 bytes for the public exponent.
653  */
654 #define PSA_KEY_EXPORT_RSA_PUBLIC_KEY_MAX_SIZE(key_bits) \
655  (PSA_KEY_EXPORT_ASN1_INTEGER_MAX_SIZE(key_bits) + 11)
656 
657 /* Maximum size of the export encoding of an RSA key pair.
658  * Assumes thatthe public exponent is less than 2^32 and that the size
659  * difference between the two primes is at most 1 bit.
660  *
661  * RSAPrivateKey ::= SEQUENCE {
662  * version Version, -- 0
663  * modulus INTEGER, -- N-bit
664  * publicExponent INTEGER, -- 32-bit
665  * privateExponent INTEGER, -- N-bit
666  * prime1 INTEGER, -- N/2-bit
667  * prime2 INTEGER, -- N/2-bit
668  * exponent1 INTEGER, -- N/2-bit
669  * exponent2 INTEGER, -- N/2-bit
670  * coefficient INTEGER, -- N/2-bit
671  * }
672  *
673  * - 4 bytes of SEQUENCE overhead;
674  * - 3 bytes of version;
675  * - 7 half-size INTEGERs plus 2 full-size INTEGERs,
676  * overapproximated as 9 half-size INTEGERS;
677  * - 7 bytes for the public exponent.
678  */
679 #define PSA_KEY_EXPORT_RSA_KEY_PAIR_MAX_SIZE(key_bits) \
680  (9 * PSA_KEY_EXPORT_ASN1_INTEGER_MAX_SIZE((key_bits) / 2 + 1) + 14)
681 
682 /* Maximum size of the export encoding of a DSA public key.
683  *
684  * SubjectPublicKeyInfo ::= SEQUENCE {
685  * algorithm AlgorithmIdentifier,
686  * subjectPublicKey BIT STRING } -- contains DSAPublicKey
687  * AlgorithmIdentifier ::= SEQUENCE {
688  * algorithm OBJECT IDENTIFIER,
689  * parameters Dss-Params } -- SEQUENCE of 3 INTEGERs
690  * DSAPublicKey ::= INTEGER -- public key, Y
691  *
692  * - 3 * 4 bytes of SEQUENCE overhead;
693  * - 1 + 1 + 7 bytes of algorithm (DSA OID);
694  * - 4 bytes of BIT STRING overhead;
695  * - 3 full-size INTEGERs (p, g, y);
696  * - 1 + 1 + 32 bytes for 1 sub-size INTEGER (q <= 256 bits).
697  */
698 #define PSA_KEY_EXPORT_DSA_PUBLIC_KEY_MAX_SIZE(key_bits) \
699  (PSA_KEY_EXPORT_ASN1_INTEGER_MAX_SIZE(key_bits) * 3 + 59)

704  * version Version, -- 0
705  * prime INTEGER, -- p
706  * subprime INTEGER, -- q
707  * generator INTEGER, -- g
708  * public INTEGER, -- y
