163  *

627 /* Maximum size of the ASN.1 encoding of an INTEGER with the specified
628  * number of bits.
629  *
630  * This definition assumes that bits <= 2^19 - 9 so that the length field
631  * is at most 3 bytes. The length of the encoding is the length of the
632  * bit string padded to a whole number of bytes plus:
633  * - 1 type byte;
634  * - 1 to 3 length bytes;
635  * - 0 to 1 bytes of leading 0 due to the sign bit.
636  */

639 /* Maximum size of the export encoding of an RSA public key.
640  * Assumes that the public exponent is less than 2^32.
641  *
642  * RSAPublicKey ::= SEQUENCE {
643  * modulus INTEGER, -- n
644  * publicExponent INTEGER } -- e
645  *
646  * - 4 bytes of SEQUENCE overhead;
647  * - n : INTEGER;
648  * - 7 bytes for the public exponent.

653 /* Maximum size of the export encoding of an RSA key pair.
654  * Assumes thatthe public exponent is less than 2^32 and that the size
655  * difference between the two primes is at most 1 bit.
656  *
657  * RSAPrivateKey ::= SEQUENCE {
658  * version Version, -- 0
659  * modulus INTEGER, -- N-bit
660  * publicExponent INTEGER, -- 32-bit
661  * privateExponent INTEGER, -- N-bit
662  * prime1 INTEGER, -- N/2-bit
663  * prime2 INTEGER, -- N/2-bit
664  * exponent1 INTEGER, -- N/2-bit
665  * exponent2 INTEGER, -- N/2-bit
666  * coefficient INTEGER, -- N/2-bit
667  * }
668  *
669  * - 4 bytes of SEQUENCE overhead;
670  * - 3 bytes of version;
671  * - 7 half-size INTEGERs plus 2 full-size INTEGERs,
672  * overapproximated as 9 half-size INTEGERS;
673  * - 7 bytes for the public exponent.
674  */
675 #define PSA_KEY_EXPORT_RSA_KEY_PAIR_MAX_SIZE(key_bits) \
676  (9 * PSA_KEY_EXPORT_ASN1_INTEGER_MAX_SIZE((key_bits) / 2 + 1) + 14)
