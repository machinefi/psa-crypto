93 /* We need to set IOTEX_PK_SIGNATURE_MAX_SIZE to the maximum signature
94  * size among the supported signature types. Do it by starting at 0,
95  * then incrementally increasing to be large enough for each supported
96  * signature mechanism.
97  *
98  * The resulting value can be 0, for example if IOTEX_ECDH_C is enabled
99  * (which allows the pk module to be included) but neither IOTEX_ECDSA_C
100  * nor IOTEX_RSA_C nor any opaque signature mechanism (PSA or RSA_ALT).
101  */
102 #define IOTEX_PK_SIGNATURE_MAX_SIZE 0
103 
104 #if(defined(IOTEX_RSA_C) || defined(IOTEX_PK_RSA_ALT_SUPPORT)) && \
105  IOTEX_MPI_MAX_SIZE > IOTEX_PK_SIGNATURE_MAX_SIZE
106  /* For RSA, the signature can be as large as the bignum module allows.
107  * For RSA_ALT, the signature size is not necessarily tied to what the
108  * bignum module can do, but in the absence of any specific setting,
109  * we use that (rsa_alt_sign_wrap in library/pk_wrap.h will check). */
110  #undef IOTEX_PK_SIGNATURE_MAX_SIZE
111  #define IOTEX_PK_SIGNATURE_MAX_SIZE IOTEX_MPI_MAX_SIZE
112 #endif
113 
114 #if defined(IOTEX_ECDSA_C) && IOTEX_ECDSA_MAX_LEN > IOTEX_PK_SIGNATURE_MAX_SIZE
115  /* For ECDSA, the ecdsa module exports a constant for the maximum
116  * signature size. */
117  #undef IOTEX_PK_SIGNATURE_MAX_SIZE
118  #define IOTEX_PK_SIGNATURE_MAX_SIZE IOTEX_ECDSA_MAX_LEN
119 #endif
120 
121 #if defined(IOTEX_USE_PSA_CRYPTO)
122  #if PSA_SIGNATURE_MAX_SIZE > IOTEX_PK_SIGNATURE_MAX_SIZE
123  /* PSA_SIGNATURE_MAX_SIZE is the maximum size of a signature made
124  * through the PSA API in the PSA representation. */
125  #undef IOTEX_PK_SIGNATURE_MAX_SIZE
126  #define IOTEX_PK_SIGNATURE_MAX_SIZE PSA_SIGNATURE_MAX_SIZE
127  #endif
128 
129  #if PSA_VENDOR_ECDSA_SIGNATURE_MAX_SIZE + 11 > IOTEX_PK_SIGNATURE_MAX_SIZE
130  /* The Mbed TLS representation is different for ECDSA signatures:
131  * PSA uses the raw concatenation of r and s,
132  * whereas Mbed TLS uses the ASN.1 representation (SEQUENCE of two INTEGERs).
133  * Add the overhead of ASN.1: up to (1+2) + 2 * (1+2+1) for the
134  * types, lengths (represented by up to 2 bytes), and potential leading
135  * zeros of the INTEGERs and the SEQUENCE. */
136  #undef IOTEX_PK_SIGNATURE_MAX_SIZE
137  #define IOTEX_PK_SIGNATURE_MAX_SIZE (PSA_VENDOR_ECDSA_SIGNATURE_MAX_SIZE + 11)
138  #endif
139 #endif /* defined(IOTEX_USE_PSA_CRYPTO) */
