227  #if !defined(IOTEX_ECP_WINDOW_SIZE)
228  /*
229  * Maximum "window" size used for point multiplication.
230  * Default: a point where higher memory usage yields diminishing performance
231  * returns.
232  * Minimum value: 2. Maximum value: 7.
233  *
234  * Result is an array of at most ( 1 << ( IOTEX_ECP_WINDOW_SIZE - 1 ) )
235  * points used for point multiplication. This value is directly tied to EC
236  * peak memory usage, so decreasing it by one should roughly cut memory usage
237  * by two (if large curves are in use).
238  *
239  * Reduction in size may reduce speed, but larger curves are impacted first.
240  * Sample performances (in ECDHE handshakes/s, with FIXED_POINT_OPTIM = 1):
241  * w-size: 6 5 4 3 2
242  * 521 145 141 135 120 97
243  * 384 214 209 198 177 146
244  * 256 320 320 303 262 226
245  * 224 475 475 453 398 342
246  * 192 640 640 633 587 476
247  */
248  #define IOTEX_ECP_WINDOW_SIZE 4
249  #endif /* IOTEX_ECP_WINDOW_SIZE */
250 
251  #if !defined(IOTEX_ECP_FIXED_POINT_OPTIM)
252  /*
253  * Trade code size for speed on fixed-point multiplication.
254  *
255  * This speeds up repeated multiplication of the generator (that is, the
256  * multiplication in ECDSA signatures, and half of the multiplications in
257  * ECDSA verification and ECDHE) by a factor roughly 3 to 4.
258  *
259  * For each n-bit Short Weierstrass curve that is enabled, this adds 4n bytes
260  * of code size if n < 384 and 8n otherwise.
261  *
262  * Change this value to 0 to reduce code size.
263  */
264  #define IOTEX_ECP_FIXED_POINT_OPTIM 1
265  #endif /* IOTEX_ECP_FIXED_POINT_OPTIM */
